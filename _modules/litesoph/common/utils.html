[docs]def create_dir(project_path):
    """ Creates project directory. """
    project_path = Path(project_path)
    parent_writeable = os.access(project_path.parent, os.W_OK)

    if not parent_writeable:
        msg = f'Permission denied creating directory: {project_path}'
        raise PermissionError(msg)

    project_path = Path(project_path)
    os.makedirs(project_path)




[docs]def get_new_directory(path:Path) -> Path:
    """Checks if the directory exists and addeds a number to name untill
    a new name is found."""
    parent = path.parent
    name = path.name
    i = 1
    while True:
        if path.exists():
            name = name + f'{i}'
            path = parent / name
            i =+ 1
        else:
            break
    return path
